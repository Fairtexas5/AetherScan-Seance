  ESP32 CSI Receiver Firmware — paste into app_main.c

#include <stdio.h>
#include <string.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "esp_wifi.h"
#include "esp_event.h"
#include "esp_log.h"
#include "nvs_flash.h"
#include "lwip/sockets.h"

#define WIFI_SSID    "CSI_MAPPER"
#define WIFI_PASS    "12345678"
#define WIFI_CHANNEL  1

static const char *TAG = "CSI_RX";
static bool wifi_connected = false;

// Called for EVERY received packet — prints CSI to serial
void csi_callback(void *ctx, wifi_csi_info_t *data) {
    if (!data || !data->buf) return;
    int8_t *buf = data->buf;
    int     len = data->len;

    printf("CSI_DATA,%llu,%d,%d,",
           (unsigned long long)esp_timer_get_time(),
           data->rx_ctrl.rssi,
           len / 2);

    for (int i = 0; i < len; i++) {
        if (i < len - 1) printf("%d,", buf[i]);
        else             printf("%d",  buf[i]);
    }
    printf("\n");
}

static void wifi_event_handler(void *arg, esp_event_base_t base,
                               int32_t id, void *data) {
    if (base == WIFI_EVENT && id == WIFI_EVENT_STA_START)
        esp_wifi_connect();
    else if (base == IP_EVENT && id == IP_EVENT_STA_GOT_IP) {
        ESP_LOGI(TAG, "Connected. CSI starting...");
        wifi_connected = true;
    } else if (base == WIFI_EVENT && id == WIFI_EVENT_STA_DISCONNECTED) {
        wifi_connected = false;
        esp_wifi_connect();
    }
}

void app_main(void) {
    esp_err_t ret = nvs_flash_init();
    if (ret == ESP_ERR_NVS_NO_FREE_PAGES ||
        ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
        nvs_flash_erase(); nvs_flash_init();
    }

    esp_netif_init();
    esp_event_loop_create_default();
    esp_netif_create_default_wifi_sta();

    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    esp_wifi_init(&cfg);

    esp_event_handler_register(WIFI_EVENT, ESP_EVENT_ANY_ID,
                               wifi_event_handler, NULL);
    esp_event_handler_register(IP_EVENT, IP_EVENT_STA_GOT_IP,
                               wifi_event_handler, NULL);

    wifi_config_t sta_cfg = {
        .sta = { .ssid = WIFI_SSID, .password = WIFI_PASS,
                 .channel = WIFI_CHANNEL }
    };
    esp_wifi_set_mode(WIFI_MODE_STA);
    esp_wifi_set_config(WIFI_IF_STA, &sta_cfg);
    esp_wifi_start();

    int retries = 0;
    while (!wifi_connected && retries < 30) {
        vTaskDelay(pdMS_TO_TICKS(500)); retries++;
    }
    if (!wifi_connected) {
        ESP_LOGE(TAG, "Failed to connect. Check SSID/password.");
        return;
    }

    // ── Enable CSI capture ──────────────────────────────────────
    wifi_csi_config_t csi_cfg = {
        .lltf_en           = true,
        .htltf_en          = true,
        .stbc_htltf2_en    = true,
        .ltf_merge_en      = true,
        .channel_filter_en = false,
        .manu_scale        = false,
    };
    esp_wifi_set_csi_config(&csi_cfg);
    esp_wifi_set_csi_rx_cb(csi_callback, NULL);
    esp_wifi_set_csi(true);

    ESP_LOGI(TAG, "CSI active. Streaming data...");
    while (1) { vTaskDelay(pdMS_TO_TICKS(10000)); }
}

Collecting CSI Data
The walk-around session — capturing data at every grid position

⚠️  IMPORTANT
CLOSE the ESP-IDF Monitor in VS Code before running the Python collection script.

Python script will crash with a 'port already in use' error.


Step 1 — Set Up the Collection Script

Open Notepad and paste the following Python script. Save it as collect_data.py on your Desktop.

  collect_data.py — save this to your Desktop

import serial, time, os

# ── CONFIGURE THESE TWO LINES ──────────────────────────────────
SERIAL_PORT      = 'COM4'    # Your ESP32 COM port from Device Manager
PACKETS_PER_POINT = 200      # Frames to collect at each position
# ────────────────────────────────────────────────────────────────

SAVE_FOLDER = 'csi_data'
os.makedirs(SAVE_FOLDER, exist_ok=True)

def collect_point(row, col, ser):
    filename = os.path.join(SAVE_FOLDER, f'r{row:02d}_c{col:02d}.csv')
    print(f'Collecting row={row} col={col} → {filename}')
    packets = []
    ser.reset_input_buffer()
    while len(packets) < PACKETS_PER_POINT:
        line = ser.readline().decode('utf-8', errors='ignore').strip()
        if line.startswith('CSI_DATA'):
            packets.append(line)
    with open(filename, 'w') as f:
        f.write('timestamp,rssi,n_subcarriers,iq_values\n')
        for pkt in packets:
            f.write(pkt + '\n')
    print(f'  Saved {len(packets)} packets.')

ser = serial.Serial(SERIAL_PORT, 115200, timeout=5)
time.sleep(2)
print('Serial connected. Starting data collection...')

rows = int(input('How many ROWS in your grid? '))
cols = int(input('How many COLS in your grid? '))

for r in range(rows):
    for c in range(cols):
        input(f'\nPlace ESP32 at row={r}, col={c}. Press Enter when ready...')
        collect_point(r, c, ser)

ser.close()
print('\nData collection complete! Files saved in: csi_data/')

Best Practices During Collection

Practice	Why it Matters
Stand 0.5m+ away from ESP32 during capture	Your body absorbs WiFi — you'd create a 'person shadow' in the data
Hold the ESP32 at the same height at every position (use a box)	Mixing heights creates inconsistent vertical data
Nobody else walks around during collection	Moving people change the RF field between measurements
Don't move the NodeMCU transmitter at all	Even 5cm of transmitter movement makes all measurements inconsistent
Close windows during collection	Outdoor weather changes the RF environment over time
Collect with furniture in its normal position	Furniture shows up in the map — this is the whole point!

SOFTWARE  ·  PHASE S-4
Installing Npcap
Windows WiFi packet capture driver for real RF sensing

💡  WHAT IS THIS?
Npcap is a Windows driver that lets your WiFi card listen to ALL radio frames in the air,
not just the ones addressed to you. This is called 'monitor mode' or passive packet capture.

Your router already broadcasts beacon frames 10 times per second in every direction.
Npcap lets Python read those frames — including the signal strength (RSSI) from each one.

This is the same driver used by Wireshark, the industry-standard network analysis tool.
It is completely safe and legal on your own network.

SOFTWARE  ·  PHASE S-7
Reading Your 3D Heatmap
Understanding what every color and pattern means

What the Animation Shows

The animation sweeps through three axes of the 3D room volume in a continuous loop:

Animation Phase	What You See	Look For
Height sweep (floor → ceiling)	Floor plan view from above. Slice moves upward.	Walls at edges always bright. Transmitter corner brightest.
Depth sweep (front → back)	Front cross-section. Slice moves toward back wall.	Floor and ceiling bright. Furniture blobs in the middle.
Width sweep (left → right)	Side cross-section. Slice moves across the room.	Both walls bright. Open center darker.

Pattern Recognition Guide

Pattern in heatmap	Physical cause
Bright band around entire edge of floor plan	All four walls — radio waves reflect strongly off flat surfaces
Extra-bright spots at floor plan corners	Wall corners = maximum reflections from two surfaces simultaneously
One corner much brighter than others	Your transmitter is in or near that corner
Dark oval in the center of the floor plan	Open floor space — clean line-of-sight signal, few reflections
Bright blob floating in the middle of a slice	A large piece of furniture (desk, wardrobe, shelving unit)
Bright horizontal band in cross-section	Floor or ceiling surface reflection
Gradient brightening toward one side	Transmitter is on that side — signal decays with distance
Asymmetric pattern left/right	Room is not empty — objects on one side create asymmetry

How to Improve Weak Results

•	More packets per position — change PACKETS_PER_POINT to 500 in collect_data.py
•	Finer grid — use 0.5m spacing instead of 1.0m for much better spatial detail
•	Capture when the room is empty — people absorb RF and create false shadows
•	Don't move while collecting — even shifting your weight changes the measurement
•	More positions near walls — wall areas benefit most from close-up measurements
